1.DECLARE
  V_TICKET_NO NUMBER;
  V_NAME VARCHAR2(20);
  V_AGE NUMBER;
  V_SEX VARCHAR2(6);
  V_FARE NUMBER;
  BEGIN
  V_TICKET_NO := &TICKET_NO;
  SELECT NAME, AGE, SEX, FARE
  INTO V_NAME, V_AGE, V_SEX, V_FARE
  FROM TICKET_DETAIL
  WHERE TICKET_NO = V_TICKET_NO;
  DBMS_OUTPUT.PUT_LINE('Name: ' || V_NAME);
  DBMS_OUTPUT.PUT_LINE('Age: ' || V_AGE);
  DBMS_OUTPUT.PUT_LINE('Sex: ' || V_SEX);
  DBMS_OUTPUT.PUT_LINE('Fare: ' || V_FARE);
  END;
  /

2.SET SERVEROUTPUT ON;
  BEGIN
  UPDATE ROUTE_DETAIL
  SET NONSTOP = 'n'
  WHERE ROUTE_ID BETWEEN 105 AND 115;
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Nonstop updated to "n" for
  ROUTE_IDs 105 to 115.');
  END;
  /
