CREATE TABLE EMP
(
ENAME CHAR(1),
EID NUMBER(3),
MID NUMBER(3),
SALARY NUMBER(5)
);

INSERT ALL
INTO EMP VALUES('P',004,NULL,5000)
INTO EMP VALUES('Q',002,NULL,6000)
INTO EMP VALUES('R',022,NULL,7000)
INTO EMP VALUES('S',010,004,8000)
INTO EMP VALUES('T',011,022,9000)
INTO EMP VALUES('U',012,002,10000)
SELECT * FROM DUAL;

CREATE TABLE JOB_SAL
(
GRADE CHAR(1),
LOWERSAL NUMBER(5),
HIGHERSAL NUMBER(5)
);

INSERT ALL
INTO JOB_SAL VALUES('A',1000,1500)
INTO JOB_SAL VALUES('B',1501,3000)
INTO JOB_SAL VALUES('C',3001,5000)
INTO JOB_SAL VALUES('D',5001,8000)
INTO JOB_SAL VALUES('E',8001,13000)
SELECT * FROM DUAL;

SELECT E.ENAME, E.SALARY, J.GRADE
FROM EMP E, JOB_SAL J
WHERE E.SALARY BETWEEN J.LOWERSAL AND J.HIGHERSAL;

SELECT * 
FROM EMP 
NATURAL JOIN JOB_SAL
WHERE EMP.SALARY=5000;

SELECT E1.ENAME, E1.EID, E1.MID
FROM EMP E1, EMP E2
WHERE E1.EID=E2.MID;

SELECT ENAME,GRADE FROM EMP CROSS JOIN JOB_SAL;

CREATE TABLE CUST_ID
(
CID VARCHAR2(5),

);

SELECT EMPLOYEE_ID.EID, EMPLOYEE_ID.CITY, CUST_ID.CID, CUST_ID.CITY
FROM EMPLOYEE_ID, CUST_ID
WHERE EMPLOYEE_ID.CITY=CUST_ID.CITY;

SELECT EMPLOYEE_ID.EID, EMPLOYEE_ID.CITY, CUST_ID.CID, CUST_ID.CITY
FROM EMPLOYEE_ID, CUST_ID
WHERE EMPLOYEE_ID.CITY=CUST_ID.CITY(+);

SELECT EMPLOYEE_ID.EID, EMPLOYEE_ID.CITY, CUST_ID.CID, CUST_ID.CITY
FROM EMPLOYEE_ID, CUST_ID
WHERE EMPLOYEE_ID.CITY(+)=CUST_ID.CITY;


SELECT EMPLOYEE_ID.CITY, CUST_ID.CITY
FROM EMPLOYEE_ID
FULL OUTER JOIN CUST_ID ON EMPLOYEE_ID.CITY=CUST_ID.CITY;
