1.SET SERVEROUTPUT ON
  DECLARE
  V_ROUTE_ID NUMBER;
  V_DISTANCE NUMBER;
  BEGIN
  V_ROUTE_ID := &ROUTE_ID;
  SELECT DISTANCE INTO V_DISTANCE
  FROM ROUTE_HEADER
  WHERE ROUTE_ID = V_ROUTE_ID;
  IF V_DISTANCE < 500 THEN

a)UPDATE ROUTE_HEADER
  SET FARE = 190.80
  WHERE ROUTE_ID = V_ROUTE_ID;
  DBMS_OUTPUT.PUT_LINE('Fare updated');

b)ELSIF V_DISTANCE >= 500 AND V_DISTANCE <= 1000 THEN
  UPDATE ROUTE_HEADER
  SET FARE = 876.98
  WHERE ROUTE_ID = V_ROUTE_ID;
  DBMS_OUTPUT.PUT_LINE('Fare updated');

c)ELSE
  DBMS_OUTPUT.PUT_LINE('Distance is more than 1000. No fare updated.');
  END IF;
  END;
  /

2.SET SERVEROUTPUT ON;
  DECLARE
  V_PLACE_ID NUMBER := 1;
  BEGIN
  UPDATE PLACE_HEADER
  SET BUS_STATION = 'n'
  WHERE PLACE_ID = V_PLACE_ID OR PLACE_ID = 6;
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Bus stations updated.');
  END;
  /
